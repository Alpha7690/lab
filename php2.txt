If (isset($_GET[‘choice’], $_GET[‘filename’])) {
    $choice = $_GET[‘choice’];
    $file = $_GET[‘filename’];
    $txt = $_GET[‘content’] ?? “”;
    Switch ($choice) {
        Case 1:
            Echo fopen($file, “w”) ? “File Created” : “Create Failed”;
            Break;
        Case 2:
            Echo file_put_contents($file, $txt) !== false ? “Written” : “Write Failed”;
            Break;
        Case 3:
            Echo file_exists($file) ? nl2br(file_get_contents($file)) : “Not Found”;
            Break;
        Case 4:
            Echo copy($file, “copy_$file”) ? “Copied” : “Copy Failed”;
            Break;
        Case 5:
            Echo unlink($file) ? “Deleted” : “Delete Failed”;
            Break;
        Default:
            Echo “<p>Invalid choice</p>”;
    }
}
