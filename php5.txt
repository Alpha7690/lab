<?php
$name=$email=$roll=$dept=$book="";
$nameErr=$emailErr=$rollErr=$deptErr=$bookErr="";
if ($_SERVER["REQUEST_METHOD"]=="POST")
{
    $name=$_POST["name"];
    $email=$_POST["email"];
    $roll=$_POST["roll"];
    $dept=$_POST["dept"];
    $book=$_POST["book"];
    if (!$name)
    {
        $nameErr="Name is required";
    }
    elseif(!preg_match("/^[a-zA-Z]+$/",$name))
    {
        $nameErr="ONLY LETTERS ALLOWED";
    }
    if (!$email)
    {
        $emailErr="Mail is required";
    }
    elseif(!filter_var($email,FILTER_VALIDATE_EMAIL))
    {
        $nameErr="INVALID EMAIL FORM";
    }
    if(!$roll)
    {
        $rollErr="roll number is required"; 
    }
    elseif (!preg_match("/^[0-9]+$/",$roll)) 
    {
        $rollErr="only numbers allowed";
    }
    if (empty($dept))
    {
        $deptErr="Department is required";
    }
    else {
        echo "$dept<br>";
    }
    if (!$book)
    {
        $bookErr="book is required";
    }
    if(!$nameErr && !$emailErr && !$rollErr)
    {
        echo "Name : $name <br> EMAIL : $email <br> roll : $roll <br> DEPT : $dept <br> BOOK : $book";
    }
    else {
        echo $nameErr;
        echo $rollErr;
        echo $emailErr;
        echo $deptErr;
        echo $bookErr;
    }
}
